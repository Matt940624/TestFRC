#include <iostream>
#include <iomanip>
#include <string.h>
#define int long long
using namespace std;

const int MAXN = 500;
int dp[MAXN][MAXN];
int pre[MAXN];
int n, a, left, right;

signed main() {
    cin >> n;
    memset( dp, 0x3f, sizeof(dp) );
    for( int i = 1 ; i <= n ; i++ ){
        cin >> a;
        pre[i] = pre[i-1] + a;
        dp[i][i] = 0;
    }
    // len = k
    for( int len = 2 ; len <= n ; len++ ){
        for( int i = 1 ; i <= n - len + 1 ; i++ ){
            int left = i, right = i + len - 1;
                // left = i, right = i + k
            for( int j = left ; j < right ; j++ ){
                dp[left][right] =
                    min( dp[left][right], dp[left][j] + dp[j+1][right] + pre[right] - pre[left-1] );
            }
        }
    }
    cout << dp[1][n];
}
